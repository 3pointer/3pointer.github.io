int findA(char *a){
    int countA = 0;
    int i = 0;
    while(a[i] != '\0'){
        if (a[i] == 'a'){
            countA ++;
        }
        i ++;
    }
    return countA;
}

int findB(char *a){
    int countB = 0;
    int i = 1;
    while(a[i] != '\0'){
        if (a[i] == 'b' && a[i - 1] == 'b'){
            countB ++;
            i ++;
        }
        i ++;
    }
    return countB;
}

bool replace(char *a){
        
    int countA = findA(a);
    int countB = findB(a);
    
    if (countA > countB)
        return false;

    int i = 1;
    while(a[i] != '\0'){
        if (a[i] == 'b' && a[i - 1] == 'b'){
            a[i - 1] = '1';
            int j = i;
            while(a[j] != '\0'){
                a[j] = a[j + 1];
                j ++;
            }
        }
        i++;
    }

    i = 1;
    while(a[i] != '\0'){
        if (a[i] == 'a'){
            a[i] = '2';
            int j = i + 1;
            while(a[j] != '\0'){
                a[j + 1] = a[j];
                j ++;
            }
            a[i + 1] = '2';
        }
        i ++;
    }
    
    return true;
}
